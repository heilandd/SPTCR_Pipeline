## Install gcc  and g++ compiler version 9 alternatives
sudo apt -y install gcc-9 g++-9

## Configure the Alternatives (suppose you have gcc/g++ 11 as current version)
sudo update-alternatives --install /usr/bin/gcc gcc /usr/bin/gcc-9 9
sudo update-alternatives --install /usr/bin/g++ g++ /usr/bin/g++-9 9

sudo update-alternatives --install /usr/bin/gcc gcc /usr/bin/gcc-11 11
sudo update-alternatives --install /usr/bin/g++ g++ /usr/bin/g++-11 11

## switch compiler from 11 to 9, execute as sudo 
sudo update-alternatives --config gcc
sudo update-alternatives --config g++

### Also make sure you have the development support files for zlb, execute as sudo:
sudo apt-get install libz-dev

## Compile Rattle
cd Rattle
./build.sh

## after compiling / if you are done using SPTCR-seq Pipeline, you can switch you gcc/g++ compiler again by doing 
sudo update-alternatives --config gcc
sudo update-alternatives --config g++
